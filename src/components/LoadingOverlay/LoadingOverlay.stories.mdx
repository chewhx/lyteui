## Fullscreen

You may fill the entire screen with the loader to prevent user interaction while something loads.

export const FullScreen = () => {
	const [show, setShow] = useState(false);
	const activate = () => {
		setShow(true);
		setTimeout(() => setShow(false), 5000);
	};
	return (
		<>
			<Button onClick={activate}>Show fullscreen loader for 5 secs</Button>
			{show && (
				<LoadingOverlay message="Connecting..." fullScreen opacity={0.8} />
			)}
		</>
	);
};
