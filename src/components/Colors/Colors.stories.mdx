## useColors hook

const colors = useColors();

colors.primary = '#122ecc';

{
  "primary": " #122ecc",

export const Colors = () => {
	const colors = useColors();
	return (
		<div className="row">
			<div className="col-4">
				{Object.entries(colors.theme).map(([name, value]) => (
					<div style={{ backgroundColor: value, padding: '1rem' }}>
						{name} - {value}
					</div>
				))}
			</div>
			<div className="col-4">
				{Object.entries(colors.palette.primary).map(([name, value]) => (
					<div style={{ backgroundColor: value, padding: '1rem' }}>
						Primary {name} - {value}
					</div>
				))}
			</div>
			<div className="col-4">
				{Object.entries(colors.palette.secondary).map(([name, value]) => (
					<div style={{ backgroundColor: value, padding: '1rem' }}>
						Secondary {name} - {value}
					</div>
				))}
			</div>
			<div className="col-4">
				{Object.entries(colors.palette.success).map(([name, value]) => (
					<div style={{ backgroundColor: value, padding: '1rem' }}>
						Success {name} - {value}
					</div>
				))}
			</div>
			<div className="col-4">
				{Object.entries(colors.palette.info).map(([name, value]) => (
					<div style={{ backgroundColor: value, padding: '1rem' }}>
						Info {name} - {value}
					</div>
				))}
			</div>
		</div>
	);
};

<Colors />
